    //String binarySearch function
    public static int binarySearchString(String[] arr, String target) {
        int low = 0;
        int high = arr.length - 1;

        while (low <= high) {
            int mid = (low + high) / 2;
            int comparisonResult = target.compareTo(arr[mid]);

            if (comparisonResult == 0) {
                return mid;
            } else if (comparisonResult < 0) {
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return -1;
    }

    public static int binarySearchInt(int[] numbers, int numberToFind){

        int low = 0;
        int high = numbers.length - 1;

        while (low <= high){
            int mid = (low + high) / 2;
            int middleNumber = numbers[mid];

            if (numberToFind == middleNumber){
                return mid;
            }

            else if (numberToFind < middleNumber){
                high = mid - 1;
            }

            else {
                low = mid + 1;
            }
        }
        return -1;
    }

    @Override
    public void actionPerformed(ActionEvent e) {



        if(e.getSource() == addproperty){

            Object selectedProperty = availableprprtylist.getSelectedValue();

            if(selectedProperty != null){
                ownedlistmodel.addElement((String) selectedProperty);
                availablelistmodel.removeElement(selectedProperty);

            }else{
                JOptionPane.showMessageDialog(this, "Please Choose before adding", "Select error", JOptionPane.ERROR_MESSAGE);
            }
        }

        else if(e.getSource() == removeproperty){

            Object selectedProperty = ownedprprtylist.getSelectedValue();

            if(selectedProperty != null){
                availablelistmodel.addElement((String) selectedProperty);
                ownedlistmodel.removeElement(selectedProperty);

            }else{
                JOptionPane.showMessageDialog(this, "Please Choose before removing", "Select error", JOptionPane.ERROR_MESSAGE);
            }
        }

        if (e.getSource() == add){
            String clientid = TclientId.getText().trim();
            String name = Tname.getText().trim();
            String contact = TcontactInfo.getText().trim();


            if (clientid.isEmpty() || name.isEmpty() || contact.isEmpty()) {
                JOptionPane.showMessageDialog(this, "Please Input First", "Input Error", JOptionPane.ERROR_MESSAGE);
                return;
            }

            if (clientMap.containsKey(clientid)) {
                JOptionPane.showMessageDialog(this, "Error: Client ID already exists!", "Id Name Error", JOptionPane.ERROR_MESSAGE);
            }

            else {
                tableModel.addRow(new Object[]{clientid, name, contact});
                clientMap.put(clientid, name);


                JOptionPane.showMessageDialog(this, "Add Succesfully", "Notification", JOptionPane.INFORMATION_MESSAGE);

                TclientId.setText("");
                Tname.setText("");
                TcontactInfo.setText("");
            }
        }

        else if (e.getSource() == delete){
            int indexSelected = table.getSelectedRow();
            if(indexSelected != -1){
                tableModel.removeRow(indexSelected);
                JOptionPane.showMessageDialog(this, "Deleted Client Info Succesfully", "Notification", JOptionPane.INFORMATION_MESSAGE);
            }
            else{
                JOptionPane.showMessageDialog(this, "Please Choose a client in the table before deleting", "Select error", JOptionPane.ERROR_MESSAGE);
            }
        }


        
        else if (e.getSource() == update){
            int selectedRow = table.getSelectedRow();

            if(selectedRow != -1){
                String clientid = TclientId.getText().trim();
                String name = Tname.getText().trim();
                String contact = TcontactInfo.getText().trim();

                if (clientid.isEmpty() || name.isEmpty() || contact.isEmpty()) {
                    JOptionPane.showMessageDialog(this, "Please fill all the fields before updating", "Input Error", JOptionPane.ERROR_MESSAGE);
                    return;
                }

                tableModel.setValueAt(clientid, selectedRow, 0);
                tableModel.setValueAt(name, selectedRow, 1);
                tableModel.setValueAt(contact, selectedRow, 2);

                JOptionPane.showMessageDialog(this, "Updated Client Info Successfully", "Notification", JOptionPane.INFORMATION_MESSAGE);

                TclientId.setText("");
                Tname.setText("");
                TcontactInfo.setText("");
            }
            else{
                JOptionPane.showMessageDialog(this, "Please Select a client from the table to update", "Select error", JOptionPane.ERROR_MESSAGE);
            }
        }


        
        else if (e.getSource() == selectClient){
            int selectedRow = table.getSelectedRow();

            if(selectedRow != -1){
                String clientid = (String) tableModel.getValueAt(selectedRow, 0);
                String name = (String) tableModel.getValueAt(selectedRow, 1);
                String contact = (String) tableModel.getValueAt(selectedRow, 2);

                TclientId.setText(clientid);
                Tname.setText(name);
                TcontactInfo.setText(contact);

                JOptionPane.showMessageDialog(this, "Client Info Loaded to Text Fields", "Notification", JOptionPane.INFORMATION_MESSAGE);
            }
            else{
                JOptionPane.showMessageDialog(this, "Please Select a client in the table", "Select error", JOptionPane.ERROR_MESSAGE);
            }
        }



    }
